#include<stdio.h>

FILE* AbreAquivo(char modo, char caminho[30]){
    FILE *arquivo;

    switch (modo){
    case 'g': //g = Gravação
        arquivo = fopen(caminho, "wt"); break;
    case 'r': // r = Leitura
        arquivo = fopen(caminho, "rt"); break;
    case 'a':
        arquivo = fopen(caminho, "a"); break;

    if (arquivo == NULL){
        printf("Arquivo não encontrado!");
        exit(0);
    }
    return arquivo;
}
}
